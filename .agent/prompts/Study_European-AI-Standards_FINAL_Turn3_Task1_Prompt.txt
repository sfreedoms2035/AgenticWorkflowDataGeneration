def timed_O_D(dataset_size: int, scale_factor: float = 1.0) -> Callable[[Callable[..., T]], Callable[..., T]]:

    """Decorator for O(D) operations with robust performance logging. REQ-PY-REG-001"""

            result = func(args, **kwargs)

            elapsed_ms = (time.perf_counter() - start) * 1000

        return wrapper

    return decorator

class DataLineageTracker(Generic[N]):

    def init(self, state: GovernanceState[N], cache_ttl_sec: float = 60.0):

        self.nodes: dict[N, dict[str, Any]] = {} # Compact node dict

        self.edges_parent: dict[N, set[N]] = {} # Adjacency list: node -> parents

        self.edges_child: dict[N, set[N]] = {} # Adjacency list: node -> children

        self.state = state

        self.ttl = cache_ttl_sec

        self._LOG = logging.getLogger("LINEAGE")


    def add_artifact(self, artifact_id: N, metadata: Any, parents: list[N] | None = None):

        """REQ-PY-REG-001, REQ-PY-REG-002: Add artifact with cycle detection."""

        if artifact_id in self.nodes:

            self._LOG.warning(f"Artifact {artifact_id} already exists. Updating metadata.")

        self.nodes[artifact_id] = metadata

        self.state.update_artifact(artifact_id, metadata)




        if parents:

            for parent in parents:

                self.add_lineage_edge(artifact_id, parent)


    def add_lineage_edge(self, child_id: N, parent_id: N):

        """REQ-PY-REG-002: Iterative edge addition with cycle detection."""

        if parent_id not in self.nodes:

            self._LOG.error(f"Cannot add edge: parent {parent_id} not found.")

            return


        # Cycle Detection using iterative DFS to avoid recursion limits. REQ-PY-REG-002

        stack: list[N] = [parent_id]

        visited: set[N] = {child_id}

        while stack:

            current = stack.pop()

            if current == child_id:

                self._LOG.error(f"Cycle detected: {parent_id} -> {child_id} is forbidden. Edge not added.")

                return

            if current not in visited:

                visited.add(current)

                # Check grandchildren if exists

                if current in self.edges_child:

                    stack.extend(self.edges_child[current])


        # Optimized edge addition with O(1) in-set existence check.

        if child_id not in self.edges_parent:

            self.edges_parent[child_id] = set()

        if parent_id not in self.edges_child:

            self.edges_child[parent_id] = set()


        self.edges_parent[child_id].add(parent_id)

        self.edges_child[parent_id].add(child_id)

        # Invalidate lineage cache for the affected child and parent chains. Simplified here to all lineage.

        self.state.lineage_cache = {} # Drastic cache invalidation on any edge change.


    def get_lineage(self, artifact_id: N) -> list[N]:

        """REQ-PY-REG-002: Traverse full lineage from target to sources."""

        # Cache logic using global TTL check. A more granular cache invalidation is complex.

        query_key = ("lineage", str(artifact_id))

        with self.state._lock:

            cached = self.state.lineage_cache.get(query_key)

            if cached:

                timestamp, result = cached

                if time.time() - timestamp < self.ttl:

                    self._LOG.debug(f"Lineage query cache hit for {artifact_id}.")

                    return result

                del self.state.lineage_cache[query_key]


        self._LOG.info(f"Computing lineage for artifact: {artifact_id} (O(N+E) worst case).")

        lineage: list[N] = []

        stack: list[N] = [artifact_id]

        visited: set[N] = set()

        while stack:

            current = stack.pop()

            if current not in visited:

                visited.add(current)

                lineage.append(current)

                # O(E) traversal using pre-computed adjacency dict for parents.

                    stack.extend(self.edges_parent[current])

            # Store in cache with current timestamp.

            self.state.lineage_cache[query_key] = (time.time(), lineage)

        return lineage
